    ```typescript
    const checkoutExpenseSheet = input.bills
      .filter((bill) => bill.responsibility === 'landlord' && ['rent_refund', 'deposit_refund'].includes(bill.type))
      .map((bill) => {
        const lease = input.leases.find((l) => l.id === bill.leaseId);
        const room = input.rooms.find((r) => r.id === bill.roomId);
        const tenant = input.tenants.find((t) => t.id === (lease?.tenantId || ''));
        const asset = input.assets.find((a) => a.id === (room?.assetId || ''));
        return {
          '房源/楼栋': asset?.name || '',
          '房号/房间': room?.name || '',
          '租客': tenant?.name || '',
          '租约期间': lease ? `${lease.startDate} ~ ${lease.endDate}` : '',
          '支出类型': bill.type === 'rent_refund' ? '余下租金' : '退还押金',
          '金额': bill.amount,
          '日期': bill.createdAt?.slice(0, 10) || '',
          '备注': bill.note || ''
        } as CheckoutExpenseDetailRow;
      });
    ```
    注意：`CheckoutExpenseDetailRow` 的字段名使用中文 key（与 schema 定义一致），需要确保 schema 中的 `CheckoutExpenseDetailRow` 有这些字段或改用 any 类型。
